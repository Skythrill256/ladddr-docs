## Props
| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `isOpen` | `boolean` | `false` | Controls modal visibility |
| `onClose` | `() => void` | | Callback for modal closure |
| `transactionHash` | `string` | `""` | Transaction hash to display |
| `sourceConfirmed` | `boolean` | `false` | Source chain confirmation status |
| `destinationConfirming` | `boolean` | `true` | Destination chain confirmation status |
| `sourceText` | `string` | `"Confirmed in source"` | Source confirmation text |
| `destinationText` | `string` | `"Confirming for destination"` | Destination confirmation text |
| `primaryColor` | `string` | `"green"` | Source chain indicator color |
| `secondaryColor` | `string` | `"blue"` | Destination chain indicator color |
| `modalClassName` | `string` | `""` | Additional modal styling classes |

## Example
```tsx
"use client"
import { TransactionConfirmation } from "@/components/transaction-confirmation";
import React, { useState } from 'react';

export default function TransactionDemo() {
  const [isOpen, setIsOpen] = useState(true);
  
  return (
    <div>
      <TransactionConfirmation
        isOpen={isOpen}
        onClose={() => setIsOpen(false)}
        transactionHash="0xabc1234567890def"
        sourceConfirmed={true}
        destinationConfirming={true}
        primaryColor="purple"
        secondaryColor="orange"
        sourceText="Transaction Confirmed on Source Chain"
        destinationText="Awaiting Confirmation on Destination Chain"
        modalClassName="custom-modal rounded-lg"
      />
    </div>
  );
}
```
